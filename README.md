curl -o- https://raw.githubusercontent.com/ruizhaogit/dotfiles/refs/heads/main/install.sh | bash

after run this install.sh
source ~/.bashrc
nvm install 20.18.1
nvm use 20.18.1
npm install -g @google/gemini-cli (optional)

in vim, run :PlugInstall
